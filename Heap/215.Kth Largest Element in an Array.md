方法一：用heap
```
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        priority_queue<int, vector<int>, greater<int>> pq;
        for(int i = 0; i < nums.size(); i++) {
            if(pq.size() == k) {
                if(nums[i] > pq.top()) {
                    pq.pop();
                    pq.push(nums[i]);
                }
            } else {
                pq.push(nums[i]);
            }
        }
        return pq.top();
    }
};
```

greater<int> is a built-in function in stl
we can use it to build a min heap
