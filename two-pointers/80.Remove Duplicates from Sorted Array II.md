```java
class Solution {
    public int removeDuplicates(int[] nums) {
        if(nums.length == 0) return 0;
        int length = 0;
        int temp = nums[0];
        int count = 0;
        
        for(int i = 0; i < nums.length; i++) {
            if(nums[i] == temp && count < 2) {
                nums[length] = temp;
                count += 1;
                length += 1;
            } else if(nums[i] != temp) {
                count = 1;
                temp = nums[i];
                nums[length] = temp;
                length += 1;
            }
        }
        nums[length -1] = temp;
        return length;
    }
}
```

```cpp
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if(nums.empty()) return 0;

        short rmCount = 0;
        bool dup = false;
        for(int i = 1; i < nums.size(); i++) {
            if(nums[i - rmCount - 1] == nums[i]) {
                if(dup) rmCount++;
                else {
                    dup = true;
                    nums[i - rmCount] = nums[i];
                }
            } else {
                dup = false;
                nums[i - rmCount] = nums[i];
            }
        }
        return nums.size() -rmCount;
    }
};
```
