## Method 1 : prefix sum + hash

```cpp
// prefix sum
class Solution {
public:
    int numSubarraysWithSum(vector<int>& nums, int goal) {
        int sum = 0;
        unordered_map<int, int> cnt;
        int res = 0;
        for (const auto& n : nums) {
            cnt[sum]++;
            sum += n;
            res += cnt[sum - goal];
        }
        return res;
    }
};
```
Time complexity: O(n)
Space complexity: O(n)


## Method 2 : prefix sun + sliding window

```cpp
class Solution {
public:
    int numSubarraysWithSum(vector<int>& nums, int goal) {
        int n = nums.size();
        int left1 = 0, left2 = 0, right = 0;
        int sum1 = 0, sum2 = 0;
        int res = 0;
        while (right < n) {
            sum1 += nums[right];
            while (left1 <= right && sum1 > goal) {
                sum1 -= nums[left1];
                left1++;
            }
            sum2 += nums[right];
            while (left2 <= right && sum2 >= goal) {
                sum2 -= nums[left2];
                left2++;
            }
            res += left2 - left1;
            right++;
        }
        return res;
    }
};
```
