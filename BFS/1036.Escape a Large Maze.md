## Method 1

看官方题解方法一
```go
type pair struct{ x, y int}
var dirs = []pair{{-1, 0}, {1, 0}, {0, -1}, {0, 1}}

func isEscapePossible(block [][]int, source []int, target []int) bool {
    const (
        blocked = -1 // 在包围圈中
        valid = 0 // 不在包围圈中
        found = 1 // 无论在不在包围圈中，但在 n(n-1)/2 步搜索的过程中经过了 target

        boundary int = 1e6
    )

    n := len(block)
    if n < 2 {
        return true
    }

    blockSet := map[pair]bool{}
    for _, b := range block {
        blockSet[pair{b[0], b[1]}] = true
    }

    check := func(start, finish []int) int {
        sx, sy := start[0], start[1]
        fx, fy := finish[0], finish[1]
        countdown := n * (n - 1) / 2

        q := []pair{{sx, sy}}
        vis := map[pair]bool {{sx, sy}: true}
        for len(q) > 0 && countdown > 0 {
            p := q[0]
            q = q[1:] // POP QUEUE
            for _, d := range dirs {
                x, y := p.x + d.x, p.y + d.y
                np := pair{x, y}      
                // move out a step
                if 0 <= x && x < boundary && 0 <= y && y < boundary && !blockSet[np] && !vis[np] {
                    if x == fx && y == fy {
                        return found
                    }
                    countdown--
                    vis[np] = true
                    q = append(q, np)
                }
            }
        }
        if countdown > 0 {
            return blocked
        }
        return valid
    }

    res := check(source, target)
    if res == found {
        return true
    } else if res == blocked {
        return false
    } else {
        if (check(target, source) == blocked) {
            return false
        }
        return true
    }
}
```
